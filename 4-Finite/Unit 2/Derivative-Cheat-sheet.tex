\subsection*{Trigonometric Derivatives}
\begin{align*}
	& \dx sin = cos & \dx cos &= -sin & \\
	& \dx tan = sec^2 & \dx cot &= -csc^2 & \\
	& \dx sec = sec * tan & \dx csc &= -csc * cot &
\end{align*}

\subsection*{Derivative Rules}
\indent \indent Power Rule
\begin{equation*}
	\dx(x^n) = n * x^{n-1}
\end{equation*}

Product Rule
\begin{equation*}
	\dx(f(x) * g(x)) = f'(x)g(x) + g'(x)f(x)
\end{equation*}

Quotient Rule
\begin{equation*}
	\dx(\frac{f(x)}{g(x)}) = \frac{ f'(x)g(x) - g'(x)f(x) }{ (g(x))^2 }
\end{equation*}

Chain Rule
\begin{equation*}
	\dx f(g(x)) = f'(g(x)) * g'(x)
\end{equation*}

Nested Chain Rule
\begin{align*}
	\dx f(g(h(x))) = & f'(g(h(x))) \\
						     & * g'(h(x)) \\
						     & * h'(x)
\end{align*}
